*Middleware~

> app.use(express.urlencoded({ extended: false }))

> app.use((req, res, next) => {
    fs.appendFile('log.txt', `${Date.now()}: ${req.method} - ${req.path}\n`,
        (err, data) => next()
     )
   })

> app.use((req, res, next) => {
    console.log("Hello from Middleware 1")
    req.myName = 'Demon'
    next();
    })

> app.use((req, res, next) => {
    console.log("Hello from Middleware 2", req.myName)
    next();
    })
